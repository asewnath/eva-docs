[docs]def read_ioda_variable(fh, group, variable, channel=None):

    # Set the variables to be read
    # ----------------------------
    if group == 'omb':
        var1 = 'ObsValue'
        var2 = 'hofx'
    elif group == 'Gsiomb':
        var1 = 'ObsValue'
        var2 = 'GsiHofX'
    elif group == 'GsiombBc':
        var1 = 'ObsValue'
        var2 = 'GsiHofXBc'
    else:
        var1 = group
        var2 = None

    # Read the data
    # -------------
    if channel is None:

        data = fh.groups[var1].variables[variable][:]

        if var2 is not None:
            data -= fh.groups[var2].variables[variable][:]

    else:

        data = fh.groups[var1].variables[variable][:, channel-1]

        if var2 is not None:
            data -= fh.groups[var2].variables[variable][:, channel-1]

    return data
